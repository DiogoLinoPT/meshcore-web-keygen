3. How it Works (Client Implementation Code)

// 1. Listen for messages from the embedded iframe
window.addEventListener('message', (event) => {
    const data = event.data;
    
    if (data && data.type === 'meshcore-api') {
        if (data.error) {
            console.error("API responded with error:", data.error);
        } else {
            console.log("Success! Received Key:", data.data);
            // Ready via data.data.publicKey & data.data.privateKey
        }
    }
});

// 2. Trigger the generation inside the iframe asynchronously
const apiFrame = document.getElementById('apiFrame');
apiFrame.contentWindow.postMessage({
    type: 'meshcore-api-request',
    prefix: '01',
    exclude: ['01A', '01B']
}, '*');
